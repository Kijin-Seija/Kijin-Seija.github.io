# 在项目中使用

查看demo

<template>
  <catalyst-list :business-config="businessConfig" />
</template>

<script>
export default {
  data() {
    return {
      businessConfig: {
        'actions': {
          'search': {
            'uri': '/api/mock/teacher/list',
          }
        },
        'columns': [
          {
            'field': {
              'type': 'raw',
              'key_name': 'id',
              'sortable': false
            },
            'label': '编号'
          },
          {
            'field': {
              'uri': 'http://www.dahai.com',
              'mode': 'page',
              'type': 'interaction',
              'key_name': 'name',
              'sortable': false,
              'show_condition': true
            },
            'label': '教师姓名'
          },
          {
            'field': {
              'type': 'raw',
              'key_name': 'age',
              'sortable': false,
              'show_condition': true
            },
            'label': '年龄'
          },
        ],
        'filters': [
          {
            'type': 'text',
            'label': '教师姓名',
            'key_name': 'name',
            'immediate': true
          },
        ]
      },
    };
  },
  provide() {
    return {
      // 列表页搜索逻辑，返回数据必须满足三个字段：list(当前页的数据条目列表)、total(数据条目总数量)、currentPage(当前页的索引)
      $mpSearch: async (url, params) => {
        let res = await getTeacherListResponse(params.page, params.page_size, params.order_by, params.sort);
        if (res.error_code === 0) {
          return res.result;
        } else {
          throw new Error(res.message);
        }
      }
    };
  }
};
</script>
