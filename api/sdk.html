# @xes/dh-catalyst

catalyst管理后台页面快速搭建框架

# 构建命令

# install dependencies
npm install

# build for production
npm run build

# 安装

npm install --save @xes/dh-catalyst

# 用法示例

// 导入
import Vue from 'vue';
import Catalyst from '@xes/dh-catalyst';
import '@xes/dh-catalyst/dist/catalyst.min.css';

Vue.use(Catalyst);

# 全局配置属性

名称 类型 必需 默认值 描述
hasPermit function(authorityCode: String): Boolean false () => true 权限控制逻辑
getNameByUrl function(url: String): String false () => '' 根据页面URL获取页面名称逻辑，用来设置页面标题
formInputComponents Object false {} 需要注册的自定义表单组件集合，属性名作为组件的名称
tableFieldComponents Object false {} 需要注册的自定义表格字段组件集合，属性名作为组件的名称

# 示例

import Vue from 'vue';
import Catalyst from '@xes/dh-catalyst';
import '@xes/dh-catalyst/dist/catalyst.min.css';
import customComponent1 from 'custom-component1';
import customComponent2 from 'custom-component2';

Vue.use(Catalyst, {
  hasPermit(authorityCode) {
    return true;
  },
  getNameByUrl(url) {
    return 'url对应页面的名称';
  },
  formInputComponents: {
    'XXXX': customComponent1,
    'XXXX': customComponent2
  },
  tableFieldComponents: {
    'XXXX': customComponent1,
    'XXXX': customComponent2
  }
});
